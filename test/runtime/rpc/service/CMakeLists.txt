project(RPCTest CXX)

set(SERVICE_SRC_DIR ${RPC_SRC_DIR}/service)
set(SERVICE_OBJ_DIR ${RPC_OBJ_DIR}/service)

include_directories(
    ./
    ${INC_DIR}
    ${SERVICE_OBJ_DIR})

set(SOURCES
    main.cpp)

set(GENERATED_SOURCES
    ${SERVICE_OBJ_DIR}/RPCTestUnit.cpp)

IMPORT_LIBRARY(comort.so)
IMPORT_LIBRARY(libservicemanager.so)

add_executable(testRPCSrv
    ${SOURCES}
    ${GENERATED_SOURCES})
target_link_libraries(testRPCSrv comort.so libservicemanager.so)
add_dependencies(testRPCSrv RPCTestUnit libservicemanager)

add_custom_command(
    OUTPUT
        ${GENERATED_SOURCES}
    COMMAND
        "$ENV{CDLC}"
        -gen
        -mode-client
        -d ${SERVICE_OBJ_DIR}
        -metadata-so "${BIN_DIR}/RPCTestUnit.so")

COPY(testRPCSrv ${SERVICE_OBJ_DIR}/testRPCSrv ${BIN_DIR})

install(FILES
    ${SERVICE_OBJ_DIR}/testRPCSrv
    DESTINATION ${BIN_DIR}
    PERMISSIONS
        OWNER_READ
        OWNER_WRITE
        OWNER_EXECUTE
        GROUP_READ
        GROUP_WRITE
        GROUP_EXECUTE
        WORLD_READ
        WORLD_EXECUTE)
