project(MacroTest CXX)

set(MACRO_SRC_DIR ${RUNTIME_SRC_DIR}/macro)
set(MACRO_OBJ_DIR ${RUNTIME_OBJ_DIR}/macro)

include_directories(
    ./
    ${INC_DIR})

set(SOURCES
    main.cpp)

IMPORT_LIBRARY(comort.so)
IMPORT_GTEST()

add_executable(testMacro
    ${SOURCES})
target_link_libraries(testMacro comort.so ${GTEST_LIBS})
add_dependencies(testMacro comort gtest gtest_main)

COPY(testMacro ${MACRO_OBJ_DIR}/testMacro ${BIN_DIR})

install(FILES
    ${MACRO_OBJ_DIR}/testMacro
    DESTINATION ${BIN_DIR}
    PERMISSIONS
        OWNER_READ
        OWNER_WRITE
        OWNER_EXECUTE
        GROUP_READ
        GROUP_WRITE
        GROUP_EXECUTE
        WORLD_READ
        WORLD_EXECUTE)
